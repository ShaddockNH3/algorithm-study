[20. 有效的括号 - 力扣（LeetCode）](https://leetcode.cn/problems/valid-parentheses/description/?envType=study-plan-v2&envId=top-100-liked)

不难，但是代码写的很史

```cpp
class Solution {
public:
    bool isValid(string s) {
        stack<char> cs;

        for(char c:s){
            if(c=='(' || c=='{' || c=='['){
                cs.push(c);
            }
            else{
                if(cs.empty()){
                    return false;
                }
                else{
                    if(c==')'){
                        if(cs.top()!='('){
                            return false;
                        }
                        else{
                            cs.pop();
                        }
                    }
                    else if(c==']'){
                        if(cs.top()!='['){
                            return false;
                        }
                        else{
                            cs.pop();
                        }
                    }
                    else if(c=='}'){
                        if(cs.top()!='{'){
                            return false;
                        }
                        else{
                            cs.pop();
                        }
                    }
                }
            }
        }

        if(cs.empty()){
            return true;
        }
        return false;
    }
};
```

其实应当构建数组来简化代码的

```cpp
class Solution {
public:
    bool isValid(string s) {
        stack<char> st;
        unordered_map<char, char> pairs = {
            {')', '('},
            {']', '['},
            {'}', '{'}
        };

        for (char c : s) {
            if (c == '(' || c == '[' || c == '{') {
                st.push(c);
            } else {
                if (st.empty() || st.top() != pairs[c]) {
                    return false;
                }
                st.pop();
            }
        }

        return st.empty();
    }
};
```
