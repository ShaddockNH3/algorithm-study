这道题初看上来像是刚学c时候的模拟题，实际上它就是一道模拟题

写起来感觉很奇怪

```cpp
#include <vector>
using namespace std;

class Solution {
public:
    vector<int> spiralOrder(vector<vector<int>>& matrix) {
        vector<int> ans;
        if (matrix.empty()) return ans;
        int m = matrix.size();    // 行
        int n = matrix[0].size(); // 列
        
        int i = 0, j = 0;
        
        while (m > 0 && n > 0) {
            // 向右移动
            for (int k = 0; k < n; k++) {
                ans.push_back(matrix[i][j]);
                j++;
            }
            j--;
            i++;
            m--;
            
            if (m == 0) break;
            
            // 向下移动
            for (int k = 0; k < m; k++) {
                ans.push_back(matrix[i][j]);
                i++;
            }
            i--;
            j--;
            n--;
            
            if (n == 0) break;
            
            // 向左移动
            for (int k = 0; k < n; k++) {
                ans.push_back(matrix[i][j]);
                j--;
            }
            j++;
            i--;
            m--;
            
            if (m == 0) break;
            
            // 向上移动
            for (int k = 0; k < m; ++k) {
                ans.push_back(matrix[i][j]);
                --i;
            }
            i++; 
            j++; 
            n--; 
        }
        
        return ans;
    }
};
```
