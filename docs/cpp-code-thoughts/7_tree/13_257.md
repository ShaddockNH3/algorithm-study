# 257 二叉树的所有路径

## 代码

```cpp
class Solution {
public:
    vector<string> binaryTreePaths(TreeNode* root) {
        vector<string> ans;
        if (root == nullptr) return ans;
        getPath(root, ans, "");
        return ans;
    }

    void getPath(TreeNode *root, vector<string>& path, string curr){
        if (!curr.empty()) {
            curr += "->";
        }
        curr += to_string(root->val);

        if(root->left == nullptr && root->right == nullptr){
            path.push_back(curr);
            return;
        }

        if(root->left) getPath(root->left, path, curr);
        if(root->right) getPath(root->right, path, curr);

    }
};
```
