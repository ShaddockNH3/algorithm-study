# 202 快乐数

## 代码

```cpp
class Solution {
public:
    bool isHappy(int n) {
        unordered_map<int,int> hasmap;
        for(;;){
            if (n==1){
                return true;
            }
            hasmap[n]+=1;
            if (hasmap[n]!=1){
                break;
            }
            n=caculate(n);
        }
        return false;
    }

    long long caculate(long long n){
        long long ans=0;
        for (;n!=0;){
            ans+=pow(n%10,2);
            n/=10;
        }
        return ans;
    }
};
```
