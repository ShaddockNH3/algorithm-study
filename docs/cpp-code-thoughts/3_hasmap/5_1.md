# 1 两数之和

## 代码

```cpp
class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        unordered_map<int,int> hasmap;

        for(int i=0;i<nums.size();i++){
            int other=target-nums[i];
            if (hasmap.count(other)){
                return vector<int> {i,hasmap[other]};
            }else{
                hasmap[nums[i]]=i;
            }
        }
        return vector<int>{-1,-1};
    }
};
```
