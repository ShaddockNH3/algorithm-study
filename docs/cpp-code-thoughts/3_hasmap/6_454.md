# 454 四数相加 II

## 代码

```cpp
class Solution {
public:
    int fourSumCount(
        vector<int>& nums1,
        vector<int>& nums2,
        vector<int>& nums3,
        vector<int>& nums4
    ) {
        unordered_map<int,int> hasmap;

        for(int i=0;i<nums1.size();i++){
            for (int j=0;j<nums2.size();j++){
                hasmap[nums1[i]+nums2[j]]+=1;
            }
        }

        int ans=0;

        for (int i=0;i<nums3.size();i++){
            for (int j=0;j<nums4.size();j++){
                int target=-(nums3[i]+nums4[j]);
                ans+=hasmap[target];
            }
        }

        return ans;
    }
};
```
