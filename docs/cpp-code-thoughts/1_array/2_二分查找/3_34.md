# 34 在排序数组中查找元素的第一个和最后一个位置

## 代码

```cpp
class Solution {
public:
    vector<int> searchRange(vector<int>& nums, int target) {
        if (nums.size() == 0) {
            return vector<int>{-1, -1};
        }
        int left_bound = BinSearchLeft(nums, target);
        if (left_bound < 0 || left_bound >= nums.size() ||
            nums[left_bound] != target) {
            return vector<int>{-1, -1};
        }
        int right_bound = BinSearchLeft(nums, target + 1);
        return vector<int>{left_bound, right_bound-1};
    }

    int BinSearchLeft(vector<int>& nums, int target) {
        int left = 0, right = nums.size() - 1;
        for (; left <= right;) {
            int mid = (right - left) / 2 + left;
            if (nums[mid] < target) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
        return right+1;
    }
};
```
