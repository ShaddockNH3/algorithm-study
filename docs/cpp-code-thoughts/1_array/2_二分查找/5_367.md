# 367 有效的完全平方数

## 代码

```cpp
class Solution {
public:
    bool isPerfectSquare(int num) {
        int x=mySqrt(num);
        return x*x==num?true:false;
    }

    int mySqrt(int x) {
        if (x==0||x==1){
            return x;
        }
        int left=1,right=x;
        for (;left<=right;){
            long long mid=(right-left)/2+left;
            if(mid*mid>x){
                right=mid-1;
            }else if (mid*mid<x){
                left=mid+1;
            }else{
                return mid;
            }
        }
        return right;
    }
};
```
