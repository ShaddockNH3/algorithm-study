# 35 搜索插入位置

## 代码

```cpp
class Solution {
public:
    int searchInsert(vector<int>& nums, int target) {
        int left,right;
        left=0;
        right=nums.size()-1;

        for(;left<=right;){
            int mid=(right-left)/2+left;
            if (nums[mid]<target){
                left=mid+1;
            }else if (nums[mid]>target){
                right=mid-1;
            }else{
                return mid;
            }
        }

        return left;
    }
};
```
