# 卡码网 58 区间和

## 代码

```cpp
#include<cstdio>

void solve(){
    int arr[100005];
    int sum[100005];

    int n;scanf("%d",&n);
    for(int i=0;i<n;i++){
        scanf("%d",&arr[i]);
        if (i==0){
            sum[i]=arr[i];
        }else{
            sum[i]=arr[i]+sum[i-1];
        }
    }

    int a,b;
    while(scanf("%d %d",&a,&b)!= EOF){
        if (a==0){
            printf("%d\n",sum[b]);
        }else{
            printf("%d\n",sum[b]-sum[a-1]);
        }
    }
}

int main(){
    solve();
}
```
