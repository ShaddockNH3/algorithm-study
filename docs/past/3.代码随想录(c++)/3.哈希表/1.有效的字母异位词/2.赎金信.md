# 2.赎金信

[383. 赎金信 - 力扣（LeetCode）][link-1]

```cpp
class Solution {
public:
    bool canConstruct(string ransomNote, string magazine) {

        if(ransomNote.size()>magazine.size()){
            return false;
        }
        int arr[26]={0};
        for(int i=0;i<ransomNote.size();i++){
            arr[ransomNote[i]-'a']-=1;
        }
        for(int i=0;i<magazine.size();i++){
            arr[magazine[i]-'a']+=1;
        }
        for(int i=0;i<26;i++){
            if(arr[i]<0){
                return false;
            }
        }
        return true;
    }
};
```
