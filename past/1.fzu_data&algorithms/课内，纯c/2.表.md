#define _CRT_SECURE_NO_WARNINGS

#include<stdio.h>
#include<math.h>
#include<stdlib.h>
#include<string.h>

#define ListItem int

typedef struct alist* List;
typedef struct alist {
	int n;
	int curr;
	int maxsize;
	ListItem *table
}Alist;

List ListInit(int size) {
	List L;
	L = (List)malloc(sizeof(List));
	L->table = (ListItem*)malloc(sizeof(ListItem) * size);
	L->maxsize = size;
	L->n = 0;
	return L;
}

int ListEmpty(List L) {
	return L->n == 0;
}

int ListLength(List L) {
	return L->n;
}

int ListLocate(int x, List L) {
	for (int i = 0; i < L->n; i++) {
		if (x == L->table[i]) {
			return ++i;
		}
	}
	return 0;
}

ListItem ListRetrieve(int k, List L) {
	return L->table[k - 1];
}

void PrintList(List L) {
	for (int i = 0; i < L->n; i++) {
		ItemShow(L->table[i]);
		if (i != L->n-1) {
			printf(" ");
		}
	}
}

void ListFree(List L) {
	free(L->table);
	free(L);
}

void ListInsert(int k,ListItem x,List L){
	for (int i = L->n - 1; i >= k; i--) {
		L->table[i + 1] = L->table[i];
	}
	L->table[k] = x;
	L->n++;
}

void ListDelete(int k, List L) {
	for (int i = k; i < L->n; i++) {
		L->table[i - 1] = L->table[i];
	}
	L->n--;
}
