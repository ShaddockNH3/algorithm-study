#include<bits/stdc++.h>
using namespace std;

void solve1(){
    string s;cin>>s;
    string word="number";

    for(int i=0;i<s.size();i++){
        if(s[i]<='9'&&s[i]>='0'){
            cout<<word;
        }
        else{
            cout<<s[i];
        }
    }
    cout<<endl;
}

void solve2(){
    string s;cin>>s;
    int oldindex=s.size()-1;

    int cnt=0;
    for(const auto&c:s){
        if(c<='9'&&c>='0'){
            cnt+=1;
        }
    }

    s.resize(s.size()+cnt*5);
    int newindex=s.size()-1;

    while(oldindex>=0){
        if(s[oldindex]>='0'&&s[oldindex]<='9'){
            s[newindex--]='r';
            s[newindex--]='e';
            s[newindex--]='b';
            s[newindex--]='m';
            s[newindex--]='u';
            s[newindex--]='n';
        }
        else{
            s[newindex--]=s[oldindex];
        }
        oldindex--;
    }

    cout<<s<<endl;
}

int main(){
    solve2();
    return 0;
}
