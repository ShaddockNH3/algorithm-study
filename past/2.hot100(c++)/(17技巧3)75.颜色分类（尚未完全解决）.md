看了提示，很快写了一个统计0，1个数的题解

```cpp
class Solution {
public:
    void sortColors(vector<int>& nums) {
        int nums_zero=0;
        int nums_one=0;
        int nums_two=0;
        for(int i=0;i<nums.size();i++){
            if(nums[i]==0){
                nums_zero+=1;
            }
            else if(nums[i]==1){
                nums_one+=1;
            }
            else{
                nums_two+=1;
            }
        }
        int i=0;
        while(nums_zero!=0){
            nums[i++]=0;
            nums_zero-=1;
        }
        while(nums_one!=0){
            nums[i++]=1;
            nums_one-=1;
        }
        while(nums_two!=0){
            nums[i++]=2;
            nums_two-=1;
        }
    }
};
```

可读性很差，不过无所谓了。
