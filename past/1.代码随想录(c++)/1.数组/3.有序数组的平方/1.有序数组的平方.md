```cpp
class Solution {
public:
    vector<int> sortedSquares(vector<int>& nums) {
        int toleft=0;
        while(toleft<nums.size()-1&&abs(nums[toleft])>=abs(nums[toleft+1])){
            toleft++;
        }
        int toright=toleft+1;
        vector<int> ans;
        while(toleft>=0&&toright<nums.size()){
            if(abs(nums[toleft])<nums[toright]){
                ans.push_back(pow(nums[toleft],2));
                toleft--;
            }
            else{
                ans.push_back(pow(nums[toright],2));
                toright++;
            }
        }
        while(toleft>=0){
            ans.push_back(pow(nums[toleft--],2));
        }
        while(toright<nums.size()){
            ans.push_back(pow(nums[toright++],2));
        }
        return ans;
    }
};
```

然后看一眼答案

答案是**双指针，从两端向中间，结果从大到小填入**，也就是倒着往前填

所以答案是不需要找到那个“最小点”，而是直接从两侧往里收。

也就是这样：

```cpp
class Solution {
public:
    vector<int> sortedSquares(vector<int>& nums) {
        int left=0,right=nums.size()-1;
        vector<int> ans(nums.size(),0);
        int k=nums.size()-1;
        while(left<=right){
            if(abs(nums[left])<abs(nums[right])){
                ans[k--]=pow(nums[right--],2);
            }
            else{
                ans[k--]=pow(nums[left++],2);
            }
        }
        return ans;
    }
};
```
