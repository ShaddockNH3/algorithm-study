[面试题 02.07. 链表相交 - 力扣（LeetCode）](https://leetcode.cn/problems/intersection-of-two-linked-lists-lcci/description/)

最简单的想法是使用unordered_set判断是否有重复，然后是以下的版本，也就是先对其，然后再判断，其实感觉仍然有点复杂就是了。

```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:

    int getsize(ListNode* head){
        ListNode* curr=head;
        int size=0;
        while(curr!=nullptr){
            curr=curr->next;
            size+=1;
        }
        return size;
    }

    ListNode *getIntersectionNode(ListNode *headA, ListNode *headB) {

        ListNode* dummyHeadA=new ListNode(0,headA);
        ListNode* dummyHeadB=new ListNode(0,headB);

        int sizeA=getsize(headA);
        int sizeB=getsize(headB);

        ListNode* currA=dummyHeadA;
        ListNode* currB=dummyHeadB;

        if(sizeA>sizeB){
            for(int i=0;i<sizeA-sizeB;i++){
                currA=currA->next;
            }
        }
        else if(sizeA<sizeB){
            for(int i=0;i<sizeB-sizeA;i++){
                currB=currB->next;
            }
        }

        while(currA!=nullptr&&currB!=nullptr){
            if(currA==currB){
                return currA;
            }
            currA=currA->next;
            currB=currB->next;
        }

        return nullptr;
    }
};
```
