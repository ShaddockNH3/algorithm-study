[242. 有效的字母异位词 - 力扣（LeetCode）](https://leetcode.cn/problems/valid-anagram/description/)

直接排序即可

然后第二个方法就是词频

```cpp
class Solution {
public:
    bool isAnagram(string s, string t) {
        if(s.size()!=t.size()){
            return false;
        }
        int arr[26]={0};
        for(int i=0;i<s.size();i++){
            arr[s[i]-'a']+=1;
            arr[t[i]-'a']-=1;
        }
        for(int i=0;i<26;i++){
            if(arr[i]!=0){
                return false;
            }
        }
        return true;
    }
};
```
