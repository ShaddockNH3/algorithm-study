[454. 四数相加 II - 力扣（LeetCode）](https://leetcode.cn/problems/4sum-ii/description/)

暴力：

```cpp
class Solution {
public:
    int fourSumCount(vector<int>& nums1, vector<int>& nums2, vector<int>& nums3, vector<int>& nums4) {
        int cnt=0;
        for(const auto&num1:nums1){
            for(const auto&num2:nums2){
                for(const auto&num3:nums3){
                    for(const auto&num4:nums4){
                        if(num1+num2+num3+num4==0){
                            cnt+=1;
                        }
                    }
                }
            }
        }
        return cnt;
    }
};
```

```cpp
class Solution {
public:
    int fourSumCount(vector<int>& nums1, vector<int>& nums2, vector<int>& nums3, vector<int>& nums4) {
        unordered_map<int,int> hasmap;
        int cnt=0;

        for(const auto& num1:nums1){
            for(const auto& num2:nums2){
                hasmap[num1+num2]+=1;
            }
        }

        for(const auto& num3:nums3){
            for(const auto& num4:nums4){
                if(hasmap.count(-num3-num4)>0){
                    cnt+=hasmap[-num3-num4];
                }
            }
        }
        return cnt;
    }
};
```
