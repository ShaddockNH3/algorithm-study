[202. 快乐数 - 力扣（LeetCode）](https://leetcode.cn/problems/happy-number/)

使用哈希表的解法

```cpp
class Solution {
public:
    int getsquare(int num){
        int squarenum=0;
        while(num!=0){
            squarenum+=(num%10)*(num%10);
            num/=10;
        }
        return squarenum;
    }

    bool isHappy(int n) {
        int num=n;
        unordered_map<int,int> hasmap;
        hasmap[num]+=1;

        while(num!=1){
            num=getsquare(num);
            hasmap[num]+=1;
            if(hasmap[num]>1){
                return false;
            }
        }
        return true;
    }
};

```
