[69. x 的平方根 - 力扣（LeetCode）](https://leetcode.cn/problems/sqrtx/)

最简单的想法：

```cpp
class Solution {
public:
    int mySqrt(int x) {
        for(long i=0;i<=x;i++){
            if(i*i<=x&&x<(i+1)*(i+1)){
                return i;
            }
        }
        return -1;
    }
};
```

此前错误的原因：

第一，边界0

第二，i的范围

那么如果使用二分查找呢？

使用二分如下

```cpp
class Solution {
public:
    int mySqrt(int x) {
        if (x == 0) {
            return 0;
        }
        if (x == 1) {
            return 1;
        }
        int left = 0, right = x / 2;
        while (left <= right) {
            long mid = left + (right - left) / 2;
            if (mid * mid <= x && x < (mid + 1) * (mid + 1)) {
                return mid;
            } else if (mid * mid < x) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
        return -1;
    }
};
```

```cpp
class Solution {
public:
    int mySqrt(int x) {
        if (x == 0) {
            return 0;
        }
        if (x == 1) {
            return 1;
        }
        int left = 0, right = x / 2;
        int ans=0;
        while (left <= right) {
            long mid = left + (right - left) / 2;
            if (mid * mid <= x) {
                ans=mid;
                left = mid + 1;
            } 
            else {
                right = mid - 1;
            }
        }
        return ans;
    }
};
```
