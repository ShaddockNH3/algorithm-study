class Solution {
public:

    bool isPerfectSquare(int num) {
        int x=num;
        if (x == 0 || x==1) {
            return true;
        }

        int left = 0, right = x / 2;
        while (left <= right) {
            long mid = left + (right - left) / 2;
            if (mid * mid <= x && x < (mid + 1) * (mid + 1)) {
                if(mid*mid==x){
                    return true;
                }
                else{
                    return false;
                }
            }
             else if (mid * mid < x) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
        return false;
    }

};
