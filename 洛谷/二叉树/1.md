```
#include<iostream>
#include<cmath>
#define ll long long
using namespace std;

ll arr[200];

int find_max(int i,int n) {
	ll max = arr[i];
	int t = i;
	for (; i < n; i++) {
		if (max < arr[i]) {
			max = arr[i];
			t = i;
		}
	}
	return t;
}

int lose_goal(int num) {
	int d = num >> 1;
	int a = find_max(0,d);
	int b = find_max(d,num);
	return a > b ? b : a;
}

int main() {
	int n; cin >> n;
	int num = pow(2, n);
	for (int i = 0; i < num; i++) {
		cin >> arr[i] ;
	}
	cout << lose_goal(num) + 1;
	return 0;
}
```
