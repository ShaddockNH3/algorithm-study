```
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode *detectCycle(ListNode *head) {
        if (head==nullptr&&head->next==nullptr){
            return nullptr;
        }

        ListNode *pred=head;
        ListNode *curr=head;

        for (;curr!=nullptr&&curr->next!=nullptr;){
            pred=pred->next;
            curr=curr->next->next;

            if (pred==curr){
                curr=head;
                for (;curr!=pred;){
                    pred=pred->next;
                    curr=curr->next;
                }
                return curr;
            }
        }
        return nullptr;
    }
};
```
