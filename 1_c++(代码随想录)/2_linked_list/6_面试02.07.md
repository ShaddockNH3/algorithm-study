/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* getIntersectionNode(ListNode* headA, ListNode* headB) {
        int la = getLength(headA);
        int lb = getLength(headB);

        ListNode* dummyHeadA = new ListNode(-1, headA);
        ListNode* tempA = dummyHeadA;

        ListNode* dummyHeadB = new ListNode(-1, headB);
        ListNode* tempB = dummyHeadB;

        if (la > lb) {
            for (int i = 0; i < la - lb; i++) {
                tempA = tempA->next;
            }
        } else {
            for (int i = 0; i < lb - la; i++) {
                tempB = tempB->next;
            }
        }

        for (; tempA && tempB;) {
            if (tempA == tempB) {
                return tempA;
            }
            tempA=tempA->next;
            tempB=tempB->next;
        }

        return nullptr;
    }

    int getLength(ListNode* head) {
        ListNode* dummyHead = new ListNode(0, head);
        ListNode* temp = dummyHead;

        int length = 0;

        for (; temp->next != nullptr;) {
            length += 1;
            temp = temp->next;
        }

        return length;
    }
};
