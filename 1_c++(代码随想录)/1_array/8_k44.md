#include <cstdio>
#include<cmath>
#include<algorithm>

using namespace std;

int getSum(int sum[], int a, int b) {
  if (a == 0) {
    return sum[b];
  }
  return sum[b] - sum[a - 1];
}

void solve() {
  int arr[105][105];
  int n, m;
  scanf("%d %d", &n, &m);
  for (int i = 0; i < n; i++) {
    for (int j = 0; j < m; j++) {
      scanf("%d", &arr[i][j]);
    }
  }
  int row[105];
  for (int i = 0; i < n; i++) {
    int sum = 0;
    for (int j = 0; j < m; j++) {
      sum += arr[i][j];
    }
    row[i] = sum;
  }
  int col[105];
  for (int j = 0; j < m; j++) {
    int sum = 0;
    for (int i = 0; i < n; i++) {
      sum += arr[i][j];
    }
    col[j] = sum;
  }

  int sumrow[105];
  for (int i = 0; i < n; i++) {
    if (i == 0) {
      sumrow[i] = row[i];
    } else {
      sumrow[i] = sumrow[i - 1] + row[i];
    }
  }

  int sumcol[105];
  for (int j = 0; j < m; j++) {
    if (j == 0) {
      sumcol[j] = col[j];
    } else {
      sumcol[j] = sumcol[j - 1] + col[j];
    }
  }

  int minrol = 0xfffffff, mincol = 0xfffffff;
  for (int i = 0; i < n - 1; i++) {
    minrol =
        min(minrol, abs(getSum(sumrow, 0, i) - getSum(sumrow, i + 1, n - 1)));
  }
  for (int j = 0; j < m - 1; j++) {
    mincol =
        min(mincol, abs(getSum(sumcol, 0, j) - getSum(sumcol, j + 1, m - 1)));
  }

  int minnum = min(mincol, minrol);

  printf("%d", minnum);
}

int main() {
  solve();
  return 0;
}
