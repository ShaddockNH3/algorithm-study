class Solution {
public:
    int removeElement(vector<int>& nums, int val) {
        if (nums.size()==0){
            return 0;
        }
        int slow=0,fast=0;
        for(;fast<nums.size();fast++){
            nums[slow]=nums[fast];
            if (nums[fast]!=val){
                slow+=1;
            }
        }
        return nums.size()-(fast-slow);
    }
};
