class Solution {
public:
    int totalFruit(vector<int>& fruits) {
        if (fruits.size() <= 2) {
            return fruits.size();
        }

        int left = 0, right = 0;
        int maxlength = 0;

        int fruit1 = fruits[0], fruit2 = fruits[0];

        for (; right < fruits.size(); right++) {
            if (fruits[right] == fruit1 || fruits[right] == fruit2) {
                if (fruits[right] != fruit1) {
                    fruit2 = fruits[right];
                }else{
                    swap(fruit1,fruit2);
                }
            } else {
                for (left = right - 1; left >= 0 && fruits[left] == fruit2;
                     left--) {
                    ;
                }
                left++;
                fruit1 = fruits[left];
                fruit2 = fruits[right];
            }
            maxlength = max(right - left + 1, maxlength);
        }

        return maxlength;
    }
};
