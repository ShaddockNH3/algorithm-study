/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int countNodes(TreeNode* root) {
        if(root==nullptr){
            return 0;
        }
        auto depth=getNum(root);
        if(depth[0]==depth[1]){
            return pow(2,depth[0])-1;
        }else{
            return 1+countNodes(root->left)+countNodes(root->right);
        }
    }

    vector<int> getNum(TreeNode *root){
        auto curr1=root;
        auto curr2=root;

        int depthLeft=0,depthRight=0;

        for(;curr1!=nullptr;){
            curr1=curr1->left;
            depthLeft+=1;
        }

        for(;curr2!=nullptr;){
            curr2=curr2->right;
            depthRight+=1;
        }

        return vector<int>{depthLeft,depthRight};
    }
};
