class Solution {
public:
    vector<int> intersection(vector<int>& nums1, vector<int>& nums2) {
        unordered_map<int, int> hasmap1;
        for (int i = 0; i < nums1.size(); i++) {
            hasmap1[nums1[i]] = 1;
        }
        
        vector<int> ans;

        for(int i=0;i<nums2.size();i++){
            if (hasmap1[nums2[i]]==1){
                ans.push_back(nums2[i]);
                hasmap1[nums2[i]]-=1;
            }
        }
        return ans;
    }
};
