#include <stdio.h>

// 快速幂算法：计算 base^exp % mod
long long mod_pow(long long base, long long exp, long long mod) {
    long long result = 1;
    base = base % mod;  // 防止 base 超过 mod
    while (exp > 0) {
        if (exp % 2 == 1)  // 如果当前指数是奇数
            result = (result * base) % mod;  // 将当前的 base 加到结果中
        base = (base * base) % mod;  // base 每次平方
        exp /= 2;  // 将指数右移一位，相当于除以 2
    }
    return result;
}

int main() {
    long long n, k;
    const long long MOD = 114514;  // 模数 114514
    
    // 输入 n 和 k
    scanf("%lld %lld", &n, &k);
    
    long long sum = 0;  // 初始化总和为 0
    
    // 计算从 1 到 n 的求和
    for (long long i = 1; i <= n; ++i) {
        // 计算 i^k % MOD
        long long power = mod_pow(i, k, MOD);
        // 累加到 sum 中，并取模
        sum = (sum + power) % MOD;
    }
    
    // 输出最终的结果
    printf("%lld\n", sum);
    
    return 0;
}
